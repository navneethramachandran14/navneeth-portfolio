Representative SQL & DAX

SQL – sales KPIs view

CREATE OR REPLACE VIEW bi.v_sales_kpis AS
SELECT d.date_date,
       SUM(s.net_amount) AS revenue,
       SUM(s.net_amount - s.cost_amount) AS gross_margin,
       SUM(s.qty) AS units
FROM retail.gold_fact_sales s
JOIN retail.silver_dim_date d ON s.date_key = d.date_key
GROUP BY d.date_date;

DAX – margin %, LY

Revenue := SUM('v_sales_kpis'[revenue])
Gross Margin := SUM('v_sales_kpis'[gross_margin])
Margin % := DIVIDE([Gross Margin], [Revenue])

Revenue LY :=
CALCULATE([Revenue], DATEADD('v_sales_kpis'[date_date], -1, YEAR))

SQL – inventory turns & OTIF

-- inventory turns = COGS / Avg Inventory
CREATE OR REPLACE VIEW bi.v_inv_turns AS
SELECT d.month,
       SUM(cogs) / NULLIF(AVG(avg_inventory),0) AS inv_turns
FROM finance.monthly_cogs c
JOIN supply.avg_inventory i USING(month)
JOIN dim_date d ON d.month = c.month
GROUP BY d.month;

-- OTIF: on-time, in-full deliveries
CREATE OR REPLACE VIEW bi.v_otif AS
SELECT DATE_TRUNC('month', delivered_at) AS month,
       100.0*AVG(CASE WHEN on_time=1 AND in_full=1 THEN 1 ELSE 0 END) AS otif_pct
FROM logistics.shipments
GROUP BY DATE_TRUNC('month', delivered_at);
